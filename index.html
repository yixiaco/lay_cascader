hover 触发子菜单-获取当前选中值

代码

<input id="demo2_1"/>
<button id="demo2_1_click" type="button" class="layui-btn layui-btn-primary layui-border-blue">获取当前选中值</button>
layui.use('layCascader', function () {
    var layCascader = layui.layCascader;
    var demo2_1 = layCascader({
      elem: '#demo2_1',
      options: options,
      props: {
        expandTrigger: 'hover'
      }
    });
    $('#demo2_1_click').click(function () {
      // 这里是单选，获取的是一个值，如果是多选则是一个数组
      var value = demo2_1.getCheckedValues();
      var node = demo2_1.getCheckedNodes();
      if (!value) {
        layer.msg('未选中值');
        return;
      }
      layer.msg('value:' + value + ',Node:' + JSON.stringify(node.data));
    });
});

选择任意一级选项

动态加载
通过lazy开启动态加载，并通过lazyload来设置加载数据源的方法。lazyload方法有两个参数，第一个参数node为当前点击的节点，第二个resolve为数据加载完成的回调(必须调用)。为了更准确的显示节点的状态，还可以对节点数据添加是否为叶子节点的标志位 (默认字段为leaf，可通过props.leaf修改)，否则会简单的以有无子节点来判断是否为叶子节点。

代码

<input id="demo14"/>
layui.use('layCascader', function () {
    var id = 0;
    // 动态加载
    layCascader({
      elem: '#demo14',
      props: {
        lazy: true,
        lazyLoad: function(node, resolve) {
          var level = node.level;
          setTimeout(function () {
            var nodes = Array.from({length: level + 2})
              .map(function (item) {
                return {
                  value: ++id,
                  label: '选项' + id,
                  leaf: level >= 1
                };
              });
            // 通过调用resolve将子节点数据返回，通知组件数据加载完成
            resolve(nodes);
          }, 1000);
        }
      }
    });
});
